## 나의 풀이
```python
# 현재 문제점 : 
# basket에서 중복되는 인형을 judge하는 단계에서
# 중복되는 인형을 터트린 후, 추가로 중복되는 case를 잡지 못하고 있음

# 그 부분만 해결되면 완벽

def solution(board, moves):
    # moves로부터 크레인을 작동시킬 위치를 불러온다.
    # 위치를 토대로 크레인을 작동시킨다.
    # 작동시킬 때에, basket 배열에 옮겨진 인형을 담는다.
    # 마지막에, basket 배열을 검사함으로서, answer를 도출해낸다.
    # 자, 그럼 함 드가자~

    # 예외처리하기 : 비어 있는 곳에 크레인을 작동시키는 경우 (헛발질)

    basket = ["init" for _ in range(len(moves))]

    for turn in range(0, len(moves)):  # turn은 몇번째 실행(턴)인지를 나타내는 변수
        for stair in range(0, len(board)):
            if board[stair][moves[turn] - 1] != 0:  # 인형의 모양이 0이 아니면
                basket[turn] = board[stair][moves[turn] - 1]
                board[stair][moves[turn] - 1] = 0
                # basket 배열에 담는다.
                break
            elif stair==len(board)-1:
                basket[turn] = "empty"
                break
            
    answer = 0
    basket.remove("empty")

    for i in range(1, len(basket)):
        try:
            if basket[i]==basket[i - 1]:
                answer += 1
                del basket[i]
                del basket[i-1]
                
        except IndexError:
            break
            
    return answer
```
